/*COMANDOS BÁSICOS DE GIT*/

git init > para inicalizar el proyecto
git status -s > nos muestra el estado de archivos
git ad nombre del arhivo  > preparar archivo para el repo
git add. > preprarar todo el proyectoi para el area de repo
git commit -m "nombre del commit" > tomar una fotografia del proyecto
git log --oneline > muestra una lista de los commits
git reset --hard id del comit > regresar al comit
git push > subir a la nube
git pull > traer los cambios de github al proyecto local
git tag nombre del tag -m "descripción" > agregar un tag al proyecto
git push --tags > subir tags

git branch nombre de la rama > crear rama
git checkout nombre de la rama > nos movemos de la rama
git branch > muestra en que rama estamos
git merge nombre de la rama > para unir el código de la rama con la rama master(main)
git branch -d nombre de la rama > eliminar rama
